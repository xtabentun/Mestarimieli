6   package mestarimieli.gui;
7   
8   import javax.swing.BoxLayout;
9   import javax.swing.JLabel;
10  import javax.swing.JPanel;
11  import mestarimieli.logiikka.Player;
12  
13  /**
14   *
15   * @author lea
16   */
17  class GuessArea1 extends JPanel {
18  
19      public GuessArea1(Player player) {
20          setLayout(new BoxLayout(this, BoxLayout.PAGE_AXIS));
21          addLabels(player);
22          add(new JLabel(" "));
23      }
24  
25      /** 
26       * Metodi poistaa käyttäjän syötteet syötehistoriasta ennenkuin lisää uuden JLabelin jokaista arvaushistorian alkiota varten.
27       * @param player 
28       */
29      
30      public void addLabels(Player player) {
31          this.removeAll();
32          for (String i : player.getGuessList()) {
33              JLabel yx = new JLabel(i);
34              add(yx);
35          }
36      }
37  
38  }
