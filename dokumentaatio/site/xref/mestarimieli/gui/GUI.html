26  /**
27   * Graafinen käyttöliittymä joka luo uuden ikkunan peliä varten. Käyttäjä voi
28   * näin pelata graafisessa ikkunassa.
29   * @author lzkosone
30   */
31  public class GUI implements Runnable {
32  
33      private JFrame frame;
34      private final Player player;
35      public Number number;
36      private final int stage;
37      public boolean won;
38      public String list;
39      public GuessArea1 guessArea;
40      public ArrayList<String> userInput;
41      /**
42       * Luokan konstruktori.
43       */
44      public GUI() {
45          player = new Player();
46          stage = 0;
47          won = false;
48          number = new Number();
49          this.userInput = player.getGuessList();
50          this.list = "empty";
51          this.guessArea = new GuessArea1(player);
52      }
53      
54      @Override
55      public void run() {
56          frame = new JFrame("Mestarimieli-peli");
57          frame.setPreferredSize(new Dimension(600, 600));
58          frame.setBackground(Color.black);
59          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
60          frame.setContentPane(setPlayerName());
61          frame.setJMenuBar(menuBar());
62          frame.revalidate();
63          frame.pack();
64          frame.setVisible(true);
65      }
66      /**
67       * Metodilla voi asettaa mikä paneeli näkyy käyttäjälle instanssin
68       * ulkopuolelta.
69       * @param p parametripaneeli.
70       */
71      public void setNewPane(JPanel p) {
72          frame.setContentPane(p);
73          frame.revalidate();
74          frame.pack();
75      }

77      private JMenuBar menuBar() {
78          JMenuBar menuBar = new JMenuBar();
79          JMenu fileMenu = new JMenu("File");
80          JMenu editMenu = new JMenu("Edit");
81          final JMenu aboutMenu = new JMenu("About");
82          final JMenu linkMenu = new JMenu("Links");
83          return menuBar;
84      }
85      /**
86       * Metodi luo uuden gamefield JPanelin, jolle asetetaan tietyt mitat ja
87       * johon asetetaan komponentteja, kuten nappeja ja tekstikenttää. Pelaaja
88       * siis käy tässä paneelissa peliä.
89       *
90       * @return metodi palauttaa JPanel gamefieldin.
91       */
92      public JPanel setGameArea() {
93          JPanel main = new JPanel();
94          main.setLayout(new BoxLayout(main, BoxLayout.LINE_AXIS));
95          JPanel gamefield = gameArea();
96          JPanel checkArea = checkArea();
97          main.add(gamefield);
98          main.add(guessArea);
99          main.add(checkArea);
100         main.revalidate();
101         return main;
102     }
103 
104     private JPanel checkArea() {
105         JPanel historyArea = new JPanel();
106         historyArea.add(new JLabel("Check \nHistory"));
107         historyArea.setLayout(new BoxLayout(historyArea, BoxLayout.PAGE_AXIS));
108         historyArea.setPreferredSize(new Dimension(100, 600));
109         historyArea.add(Box.createRigidArea(new Dimension(100, 600)));
110         historyArea.setBackground(Color.black);
111         historyArea.setBorder(BorderFactory.createLineBorder(Color.GREEN, 8));
112         return historyArea;
113     }
114 
115     private JPanel gameArea() {
116         JPanel gamefield = new JPanel();
117         gamefield.setLayout(new BoxLayout(gamefield, BoxLayout.PAGE_AXIS));
118         gamefield.setBorder(BorderFactory.createLineBorder(Color.GREEN, 8));
119         gamefield.add(Box.createVerticalGlue());
120         gamefield.setBackground(Color.black);
121         JLabel question = new JLabel("Type number: ");
122         question.setFont(new Font("Comic Sans MS", Font.BOLD, 26));
123         JTextField guess = new JTextField();
124         guess.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
125         JButton submit = new JButton("Submit");
126         submit.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
127         JLabel hint = new JLabel("Ganbatte!");
128         hint.setFont(new Font("Comic Sans MS", Font.BOLD, 26));
129         gamefield.add(question);
130         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
131         gamefield.add(guess);
132         gamefield.add(Box.createRigidArea(new Dimension(100, 20)));
133         gamefield.add(submit);
134         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
135         gamefield.add(hint);
136         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
137         submit.addActionListener(new GuessListener(guess, number, this, player, hint));
138         return gamefield;
139     }
140     /**
141      * Metodi tuottaa ensimmäisen kentän peliin elementteineen. Siinä kysytään
142      * pelaajan nimeä.
143      * @return Metodi palauttaa gamefieldin.
144      */
145     public JPanel setPlayerName() {
146         JPanel gamefield = new JPanel();
147         gamefield.setLayout(new BoxLayout(gamefield, BoxLayout.PAGE_AXIS));
148         gamefield.setBorder(BorderFactory.createLineBorder(Color.GREEN, 8));
149         gamefield.add(Box.createHorizontalGlue());
150         gamefield.setBackground(Color.black);
151         JLabel question = new JLabel("Who are you?");
152         question.setFont(new Font("Comic Sans MS", Font.BOLD, 26));
153         JTextField name = new JTextField();
154         name.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
155         JButton submit = new JButton("Submit");
156         submit.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
157         submit.addActionListener(new NameListener(name, player, this));
158         gamefield.add(question);
159         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
160         gamefield.add(name);
161         gamefield.add(Box.createRigidArea(new Dimension(100, 20)));
162         gamefield.add(submit);
163         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
164         return gamefield;
165     }
166     /**
167      * Metodi luo JPanel-näkymän käyttäjälle, jossa kysytään koodin pituutta.
168      * @return palauttaa gamefieldin.
169      */
170     public JPanel setNumber() {
171         JPanel gamefield = new JPanel();
172         gamefield.setLayout(new BoxLayout(gamefield, BoxLayout.PAGE_AXIS));
173         gamefield.setBorder(BorderFactory.createLineBorder(Color.green, 8));
174         gamefield.add(Box.createHorizontalGlue());
175         gamefield.setBackground(Color.black);
176         JTextArea textbox = new JTextArea();
177         gamefield.setLayout(new BoxLayout(gamefield, BoxLayout.PAGE_AXIS));
178         JLabel question = new JLabel(player.getName() + ", how long will be your quest? ");
179         question.setFont(new Font("Comic Sans MS", Font.BOLD, 26));
180         JTextField length = new JTextField();
181         length.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
182         gamefield.add(length);
183         JButton submit = new JButton("Submit");
184         JLabel hint = new JLabel();
185         submit.setFont(new Font("Comic Sans MS", Font.BOLD, 16));
186         submit.addActionListener(new LengthListener(length, this, number, hint));
187         gamefield.add(question);
188         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
189         gamefield.add(length);
190         gamefield.add(Box.createRigidArea(new Dimension(100, 20)));
191         gamefield.add(submit);
192         gamefield.add(Box.createRigidArea(new Dimension(100, 100)));
193         gamefield.add(hint);
194         return gamefield;
195     }

197         return frame;
