6   package ui;
7   
8   import java.io.BufferedReader;
9   import java.io.InputStreamReader;
10  import static java.lang.System.exit;
11  import mestarimieli.logiikka.Number;
12  import mestarimieli.logiikka.Player;
13  
14  /**
15   *
16   * @author lea
17   */
18  public class UI {
19  
20      public void echo(String text) {
21          System.out.print(text);
22      }
23  
24      BufferedReader input = new BufferedReader(new InputStreamReader(System.in));
25      private boolean won = false;
26  
27      public void run() throws Exception {
28  
29          echo("Who are you? ");
30          String name = input.readLine();
31  
32          Player player = new Player(name);
33  
34          echo(name + ", how long will be your quest? ");
35          String x = input.readLine();
36          int answerLength = Integer.parseInt(x);
37  
38          Number number = new Number(answerLength) {
39          };
40  
41          String answer = number.getAnswer();
42          String guess;
43          int[] retval;
44  
45          while (true) {
46  
47              if (won) {
48                  echo("Do you want to continue? ( y/n)\n");
49                  String cont = input.readLine();
50                  if (cont.equals("n")) {
51                      exit(0);
52                  }
53                  if (!cont.equals("y")) {
54                      echo("Not an answer!");
55                      continue;
56                  } else { won = false;}
57              }
58              player.guessesGrow();
59              echo(answer + "Type number: ");
60  
61              guess = input.readLine();
62  
63              if (!number.guessLength(guess)) {
64                  echo("Wrong length\n");
65                  continue;
66              }
67  
68              retval = number.checkAnswer(guess);
69              int blacks = retval[0];
70              int whites = retval[1];
71  //            echo("Blacks " + blacks + " Whites " + whites + "\n");
72              if (blacks == answerLength) {
73                  echo("You win teh game! Number of guesses " + player.getGuesses());
74                  won = true;
75                  continue;
76              } else {
77                  echo("Blacks " + blacks + " Whites " + whites + "\n");
78              }
79  //            echo("Bulls "+  bulls + " Cows " + cows + "\n");
80  
81          }
82  
83      }
84  
85  }
